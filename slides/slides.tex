\section{Post mortem analysis}
\begin{frame}
 \transdissolve
 \frametitle{MPI}
 \begin{itemize}
  \item Principle - splitting the workload between several computers and/or cores
  \item Problem - password cracker
  \item boost-mpi-python
  \item 104 sec to check $\sim$40 million password on a single core
  \item 28 sec to check $\sim$40 million password on 2 computers with a total of 4 cores
  \item Equal programming effort, better performance on more than 1 core 
 \end{itemize}
\end{frame}
